1. two sum

```javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function (nums, target) {
  for (let i = 0; i < nums.length; i++) {
    target = target - nums[i];
    const index = nums.indexOf(target);
    const nIndex = nums.indexOf(target, i + 1);

    if (nIndex === -1 && i === index) {
      target = target + nums[i];
      continue;
    }
    if (nIndex > i) {
      nums = [i, nIndex];
    } else {
      target = target + nums[i];
    }
  }
  return nums;
};
```

7. reverse integer

```javascript
/**
 * @param {number} x
 * @return {number}
 */
var reverse = function (x) {
  let a = x.toString();
  a = a.split("");

  let b = [];

  for (let i = 0; i < a.length; i++) {
    b[i] = a[a.length - 1 - i];
  }
  if (b[b.length - 1] === "-") {
    b.unshift(b.pop());
  }
  b = b.join("");

  if (b[0] == 0 && b.length > 1) {
    b = b.slice(1);
  }

  b = Number.parseInt(b);
  if (b > Math.pow(2, 31) - 1 || b < -Math.pow(2, 31)) return 0;
  return b;
};
```

14. longest common prefix
    그지같은 코드 짰음

```javascript
let firstLetter = "";

strs = strs.sort((a, b) => {
  return a.length - b.length;
});
firstLetter = strs[0];

strs = strs.map((s) => s.slice(0, firstLetter.length));
let i = 0;
if (!strs[0]) return "";

while (i < strs[0].length) {
  if (!strs.every((s) => s[i] === strs[0][i])) {
    firstLetter = strs[0].slice(0, i);

    return firstLetter;
  }
  firstLetter = strs[0].slice(0, i + 1);
  i++;
}

return firstLetter;
```
